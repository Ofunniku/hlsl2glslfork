struct v2f {
    highp vec4 pos;
    highp vec4 color;
};
struct a2v {
    highp vec4 pos;
};
uniform highp mat4 mvp;
highp vec4 ComputeColor( in highp vec4 pos );
v2f xlat_main( in a2v v );
highp vec4 ComputeColor( in highp vec4 pos ) {
    pos *= highp vec4( 1.00000, -1.00000, -1.00000, 1.00000);
    pos.xyz  *= 0.300000;
    return pos;
}
v2f xlat_main( in a2v v ) {
    highp vec4 pos;
    v2f o;
    pos = ( mvp * v.pos );
    o.color = ComputeColor( pos);
    o.pos = pos;
    return o;
}
varying highp vec4 xlv_COLOR;
void main() {
    v2f xl_retval;
    a2v xlt_v;
    xlt_v.pos = highp vec4( gl_Vertex);
    xl_retval = xlat_main( xlt_v);
    gl_Position = highp vec4( xl_retval.pos);
    xlv_COLOR = highp vec4( xl_retval.color);
}
