struct v2f {
    highp vec4 pos;
    highp vec2 uv;
    highp vec3 ray;
};
struct appdata {
    highp vec4 vertex;
    highp vec2 texcoord;
    highp vec3 texcoord1;
};
uniform highp mat4 glstate_matrix_mvp;
v2f xlat_main( in appdata v );
v2f xlat_main( in appdata v ) {
    v2f o;
    o.pos = ( glstate_matrix_mvp * v.vertex );
    o.uv = v.texcoord;
    o.ray = v.texcoord1;
    return o;
}
varying highp vec2 xlv_TEXCOORD0;
varying highp vec3 xlv_TEXCOORD1;
void main() {
    v2f xl_retval;
    appdata xlt_v;
    xlt_v.vertex = highp vec4( gl_Vertex);
    xlt_v.texcoord = highp vec2( gl_MultiTexCoord0);
    xlt_v.texcoord1 = highp vec3( gl_MultiTexCoord1);
    xl_retval = xlat_main( xlt_v);
    gl_Position = highp vec4( xl_retval.pos);
    xlv_TEXCOORD0 = highp vec2( xl_retval.uv);
    xlv_TEXCOORD1 = highp vec3( xl_retval.ray);
}
