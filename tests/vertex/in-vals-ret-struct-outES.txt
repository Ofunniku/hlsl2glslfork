struct v2f {
    highp vec4 pos;
    highp float fog;
    highp vec2 uv;
    highp vec4 color;
};
uniform highp mat4 mvp;
v2f xlat_main( in highp vec4 pos, in highp vec2 uv, in highp vec3 normal, in highp vec4 color );
v2f xlat_main( in highp vec4 pos, in highp vec2 uv, in highp vec3 normal, in highp vec4 color ) {
    v2f o;
    o.pos = ( mvp * pos );
    o.color = color;
    o.color.xyz  += normal;
    o.uv = uv;
    return o;
}
varying highp float xlv_FOG;
varying highp vec2 xlv_TEXCOORD0;
varying highp vec4 xlv_COLOR;
void main() {
    v2f xl_retval;
    xl_retval = xlat_main( highp vec4(gl_Vertex), highp vec2(gl_MultiTexCoord0), highp vec3(gl_Normal), highp vec4(gl_Color));
    gl_Position = highp vec4( xl_retval.pos);
    xlv_FOG = highp float( xl_retval.fog);
    xlv_TEXCOORD0 = highp vec2( xl_retval.uv);
    xlv_COLOR = highp vec4( xl_retval.color);
}
