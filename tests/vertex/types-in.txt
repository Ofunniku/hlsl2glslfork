float4x4 mvp;
half4x4 hmat4;
half3x3 hmat3;
half2x2 hmat2;

struct a2v {
	float4 pos : POSITION;
	float2 uv : TEXCOORD0;
	half3 normal : NORMAL;
	fixed4 color : COLOR;
};

struct v2f {
	float4 pos : POSITION;
	float fog : FOG;
	float2 uv : TEXCOORD0;
	float4 color : COLOR;
};

v2f main (a2v v)
{
	v2f o;
	o.pos = mul (mvp, v.pos);
	o.pos += mul (hmat4, v.pos);
	o.pos.xyz += mul (hmat3, v.pos.xyz);
	o.pos.xy += mul (hmat2, v.pos.xy);
	o.color = v.color;
	o.color.rgb += v.normal;
	o.uv = v.uv;
	
	float dx = v.pos.x;
	half dy = v.pos.y;
	fixed dz = v.pos.z;
	
	return o;
}
