uniform highp mat4 mvp;
void xlat_main( in highp vec4 pos, in highp vec2 uv, in highp vec3 normal, in highp vec4 color, out highp vec4 opos, out highp float ofog, out highp vec2 ouv, out highp vec4 ocolor );
void xlat_main( in highp vec4 pos, in highp vec2 uv, in highp vec3 normal, in highp vec4 color, out highp vec4 opos, out highp float ofog, out highp vec2 ouv, out highp vec4 ocolor ) {
    opos = ( mvp * pos );
    ocolor = color;
    ocolor.xyz  += normal;
    ouv = uv;
}
varying highp float xlv_FOG;
varying highp vec2 xlv_TEXCOORD0;
varying highp vec4 xlv_COLOR;
void main() {
    highp vec4 xlt_opos;
    highp float xlt_ofog;
    highp vec2 xlt_ouv;
    highp vec4 xlt_ocolor;
    xlat_main( highp vec4(gl_Vertex), highp vec2(gl_MultiTexCoord0), highp vec3(gl_Normal), highp vec4(gl_Color), xlt_opos, xlt_ofog, xlt_ouv, xlt_ocolor);
    gl_Position = highp vec4( xlt_opos);
    xlv_FOG = highp float( xlt_ofog);
    xlv_TEXCOORD0 = highp vec2( xlt_ouv);
    xlv_COLOR = highp vec4( xlt_ocolor);
}
